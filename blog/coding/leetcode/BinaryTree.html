# 二叉树的遍历

前序遍历：根节点->左子节点->右子节点
中序遍历：左子节点->根节点->右子节点
后序遍历：左子节点->右子节点->根节点
层序遍历：逐层从上到下，从左到右遍历所有节点

# 二叉树的前序遍历

递归

const preorderTraversal = (root) => {
    let res = [];
    preorder(root, res);
    return res;
}
const preorder = (root, res) => {
    if (root == null) {
        return;
    }
    res.push(root.val);
    preorder(root.left, res);
    preorder(root.right, res);
}

迭代

const postorderTraversal  = (root) => {
    let res = [];
    let stack = [root];
    while (stack.length > 0) {
        let node = stack.pop();
        if (node != null) {
            res.push(node.val);
            stack.push(node.right);
            stack.push(node.left);
        }
    }
    return res;
} 

// 写法二：
const postorderTraversal2 = function(root) {
    let res = [];
    if (root == null) {
        return res;
    }

    let stack = [];
    let node = root;
    while (stack.length > 0 || node != null) {
        while (node != null) {
            res.push(node.val);
            stack.push(node);
            node = node.left;
        }
        node = stack.pop();
        node = node.right;
    }
    return res;
}

# 二叉树的层序遍历

BFS

var levelOrder = function(root) {
    let res = [];
    if (root == null) {
        return [];
    }
    const queue = [];
    queue.push(root);
    while (queue.length !== 0) {
        const currLevelSize = queue.length;
        res.push([]);
        for (let i = 1; i <= currLevelSize; ++i) {
            const node = queue.shift();
            res[res.length - 1].push(node.val);
            if (node.left) {
                queue.push(node.left);
            }
            if (node.right) {
                queue.push(node.right);
            }
        }
    }
    return res;
};

DFS

var levelOrder = function(root) {
    if (root == null) {
        return [];
    }
    let res = [[]];
    DFS(1, root, res);
    return res;
};
const DFS = function(index, root, list) {
    if (list.length < index) {
        list.push([]);
    }
    list[index - 1].push(root.val);
    if (root.left != null) {
        DFS(index + 1, root.left, list);
    }
    if (root.right != null) {
        DFS(index + 1, root.right, list);
    }
}

# 二叉树的最大深度

二叉树的深度为根结点到最远叶子节点的最长路径的层数 递归

var maxDepth = function(root) {
    if (!root) return 0;
    let left = maxDepth(root.left);
    let right = maxDepth(root.right);
    return Math.max(left, right) + 1;
}

BFS

function maxDepth(root) {
    if (root == null) return 0;
    let deque = [];
    deque.push(root);
    let count = 0;
    while (deque.length > 0) {
        let size = deque.length;
        for (let i = 0; i < size; i++)
            let node = deque.pop();
            if (node.left != null) {
                deque.push(node.left);
            }
            if (node.right != null) {
                deque.push(node.right);
            }
        }
        count++;
    }
    return count;
}

DFS

function maxDepth(root) {
    if (root == null) return 0; 
    let count = 0;
    let stack = [];
    let depthStack = [];

    stack.push(root);
    depthStack.push(1);

    while(stack.length > 0) {
        let node = stack.pop();
        let depth = depthStack.pop();
        maxDepth = Math.max(maxDepth, depth);

        if (node.right != null) {
            stack.push(node.right);
            depthStack.push(depth + 1);
        }
        if (node.left != null) {
            stack.push(node.left);
            depthStack.push(depth + 1);
        }
    }
    return maxDepth;   
}

# 对称二叉树

function isSymmetric(root) {
    if (!root) return;
    return isMirror(root, root);
}
function isMirror(t1, t2) {
    if (t1 == null && t2 == null) return true;
    if (t1 == null || t2 == null) return false;
    return t1.val == t2.val && isMirror(t1.left, t2.right) && isMirror(t1.right, t2.left);
}

# 路径总和

给定二叉树的根节点root和一个目标和的整数targetSum，判断树中是否存在根节点到叶子节点的路径上，所有节点值相加等于目标和targetSum，存在返回true，否则返回false

// 输入 root = [5,4,8,11,null,13,4,7,2,null,null,null,1], targetSum = 22
// 输出 true;
var hasPathSum = (root, targetSum) => {
    if (root == null) return false;
    if (root.left == null && root.right == null) return root.val == targetSum;
    return hasPathSum(root.left, targetSum - root.val) || hasPathSum(root.right, targetSum - root.val);
}
