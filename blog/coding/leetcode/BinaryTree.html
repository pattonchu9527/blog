# 二叉树的遍历

前序遍历：根节点->左子节点->右子节点
中序遍历：左子节点->根节点->右子节点
后序遍历：左子节点->右子节点->根节点
层序遍历：逐层从上到下，从左到右遍历所有节点

# 二叉树的前序遍历

递归

const preorderTraversal = (root) => {
    let res = [];
    preorder(root, res);
    return res;
}
const preorder = (root, res) => {
    if (root == null) {
        return;
    }
    res.push(root.val);
    preorder(root.left, res);
    preorder(root.right, res);
}

const postorderTraversal  = (root) => {
    let res = [];
    let stack = [root];
    while (stack.length > 0) {
        let node = stack.pop();
        if (node != null) {
            res.push(node.val);
            stack.push(node.right);
            stack.push(node.left);
        }
    }
    return res;
} 

// 写法二：
const postorderTraversal2 = function(root) {
    let res = [];
    if (root == null) {
        return res;
    }

    let stack = [];
    let node = root;
    while (stack.length > 0 || node != null) {
        while (node != null) {
            res.push(node.val);
            stack.push(node);
            node = node.left;
        }
        node = stack.pop();
        node = node.right;
    }
    return res;
}

const postorder = function(node, res) {
    if (node == null) {
        return;
    }
    postorder(node.left, res);
    postorder(node.right, res);

# 二叉树的层序遍历

BFS

var levelOrder = function(root) {
    let res = [];
    if (root == null) {
        return [];
    }
    const queue = [];
    queue.push(root);
    while (queue.length !== 0) {
        const currLevelSize = queue.length;
        res.push([]);
        for (let i = 1; i <= currLevelSize; ++i) {
            const node = queue.shift();
            res[res.length - 1].push(node.val);
            if (node.left) {
                queue.push(node.left);
            }
            if (node.right) {
                queue.push(node.right);
            }
        }
    }
    return res;
};

DFS

var levelOrder = function(root) {
    if (root == null) {
        return [];
    }
    let res = [[]];
    DFS(1, root, res);
    return res;
};
const DFS = function(index, root, list) {
    if (list.length < index) {
        list.push([]);
    }
    list[index - 1].push(root.val);
    if (root.left != null) {
        DFS(index + 1, root.left, list);
    }
    if (root.right != null) {
        DFS(index + 1, root.right, list);
    }
}
